\section{Prender Leds}
\begin{lstlisting}[caption=Prender leds basicos]
// C++ code
//
int p = 13;
int p2 = 12;

void setup(){
  pinMode(p, OUTPUT);
  pinMode(p2, OUTPUT);
}

void loop() {
  digitalWrite(p2, LOW);
  digitalWrite(p, HIGH);
  delay(500); // Wait for 1000 millisecond(s)
  digitalWrite(p2, HIGH);
  digitalWrite(p, LOW);
  delay(500); // Wait for 1000 millisecond(s)
}

\end{lstlisting}
