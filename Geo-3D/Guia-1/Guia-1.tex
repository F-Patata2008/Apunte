\section{Traslaciones}
\thispagestyle{fancy}

\subsection{Ejercicio 1}
\textbf{Problema:} ¿A qué traslación equivale la rotación de un punto $A(x_0,y_0)$, en un ángulo de 45° con respecto al origen?

\textbf{Solución:} Una rotación transforma un punto $A$ en un punto $A'$. La traslación equivalente es aquella definida por el vector que une el punto original con el punto final, es decir, $\vec{T} = A' - A$.
Primero, calculamos las coordenadas del punto rotado $A'(x_1, y_1)$ usando la fórmula de rotación:
\begin{align*}
    x_1 &= x_0\cos(45^\circ) - y_0\sin(45^\circ) = x_0\frac{\sqrt{2}}{2} - y_0\frac{\sqrt{2}}{2} = \frac{\sqrt{2}}{2}(x_0 - y_0) \\
    y_1 &= x_0\sin(45^\circ) + y_0\cos(45^\circ) = x_0\frac{\sqrt{2}}{2} + y_0\frac{\sqrt{2}}{2} = \frac{\sqrt{2}}{2}(x_0 + y_0)
\end{align*}
A continuación, calculamos el vector de traslación $\vec{T} = (x_1 - x_0, y_1 - y_0)$ restando las coordenadas originales a las nuevas.
\begin{align*}
    T_x &= \frac{\sqrt{2}}{2}(x_0 - y_0) - x_0 = x_0\left(\frac{\sqrt{2}}{2}-1\right) - y_0\frac{\sqrt{2}}{2} \\
    T_y &= \frac{\sqrt{2}}{2}(x_0 + y_0) - y_0 = x_0\frac{\sqrt{2}}{2} + y_0\left(\frac{\sqrt{2}}{2}-1\right)
\end{align*}
\paragraph{Respuesta:} El vector de traslación es $\vec{T} = \left(x_0\left(\frac{\sqrt{2}}{2}-1\right) - y_0\frac{\sqrt{2}}{2}, x_0\frac{\sqrt{2}}{2} + y_0\left(\frac{\sqrt{2}}{2}-1\right)\right)$.

\subsection{Ejercicio 2}
\textbf{Problema:} Determinar el vector total $\vec{T} = \sum_{i=1}^{\infty} \vec{T_i}$ de los desplazamientos de Aquiles.

\textbf{Solución:} El vector total es la suma de una serie infinita de vectores. La suma de las componentes 'y' es trivialmente 0. La suma de las componentes 'x' constituye una serie geométrica infinita:
\[ X = 10 + 1 + \frac{1}{10} + \frac{1}{100} + \dots \]
Esta serie tiene un primer término $a=10$ y una razón común $r=1/10$. Dado que $|r| < 1$, la suma converge y se puede calcular con la fórmula $S = \frac{a}{1 - r}$.
\[ X = \frac{10}{1 - \frac{1}{10}} = \frac{10}{\frac{9}{10}} = 10 \cdot \frac{10}{9} = \frac{100}{9} \]
\paragraph{Respuesta:} El vector total de traslación es $\vec{T} = \left(\frac{100}{9}, 0\right)$.

\subsection{Ejercicio 3}
\textbf{Problema:} Determine las coordenadas de la imagen A' de un punto $A(x_0, y_0)$, al trasladarlo de acuerdo al vector $\vec{T}(r, \theta)$, con $\theta \in [0, \pi/2]$.

\textbf{Solución:} Un vector dado en forma polar $(r, \theta)$ se puede expresar en coordenadas cartesianas como $(r\cos\theta, r\sin\theta)$. La traslación de un punto $A$ por un vector $\vec{T}$ se calcula como la suma $A' = A + \vec{T}$.
\[ A'(x', y') = (x_0, y_0) + (r\cos\theta, r\sin\theta) \]
\paragraph{Respuesta:} Las coordenadas de la imagen son $A' = (x_0 + r\cos\theta, y_0 + r\sin\theta)$.

\subsection{Ejercicio 4}
\textbf{Problema:} Utilizar el resultado anterior para determinar la imagen A' de los siguientes pares de puntos y vectores.

\end{enumerate}

\subsection{Ejercicio 6}
\textbf{Problema:} Determine el ángulo del vector traslación que lleva un punto A a un punto B.
\textbf{Solución:} El vector de traslación se encuentra restando el punto de destino menos el de origen: $\vec{T} = B - A$. El ángulo se obtiene con $\theta = \arctan(T_y/T_x)$, ajustando el cuadrante según los signos de $T_x$ y $T_y$.
\begin{enumerate}[label=\alph*)]
    \item $A(-4,1)$ y $B(-1,5)$. $\vec{T} = (-1 - (-4), 5 - 1) = (3, 4)$. El vector está en el primer cuadrante.
    \[ \theta = \arctan\left(\frac{4}{3}\right) \approx \mathbf{53.13^\circ} \]
    \item $A(1,0)$ y $B(-3,-3)$. $\vec{T} = (-3 - 1, -3 - 0) = (-4, -3)$. El vector está en el tercer cuadrante.
    \[ \theta = 180^\circ + \arctan\left(\frac{-3}{-4}\right) \approx 180^\circ + 36.87^\circ = \mathbf{216.87^\circ} \]
\end{enumerate}

\section{Rotaciones}

\subsection{Ejercicio 7}
\textbf{Problema:} Determinar las coordenadas de A' al rotar A con respecto a C en un ángulo $\theta$.
\textbf{Solución:} Se utiliza la fórmula de rotación en torno a un punto arbitrario $C(h,k)$:
$A' = ((x-h)\cos\theta - (y-k)\sin\theta + h, (x-h)\sin\theta + (y-k)\cos\theta + k)$.
\begin{enumerate}[label=\alph*)]
    \item $A(5,1), C(-1,4), \theta=135^\circ$. Calculamos las coordenadas relativas: $(x-h)=6, (y-k)=-3$.
    \begin{align*}
        x' &= 6\cos(135^\circ) - (-3)\sin(135^\circ) - 1 = 6(-\frac{\sqrt{2}}{2}) + 3(\frac{\sqrt{2}}{2}) - 1 = -1 - \frac{3\sqrt{2}}{2} \\
        y' &= 6\sin(135^\circ) + (-3)\cos(135^\circ) + 4 = 6(\frac{\sqrt{2}}{2}) - 3(-\frac{\sqrt{2}}{2}) + 4 = 4 + \frac{9\sqrt{2}}{2}
    \end{align*}
    \paragraph{Respuesta:} $\mathbf{A'\left( -1-\frac{3\sqrt{2}}{2}, 4+\frac{9\sqrt{2}}{2} \right)}$.
    
    \item $A(-10,14), C(2,-2), \theta=15^\circ$. Coordenadas relativas: $(x-h)=-12, (y-k)=16$. Usamos $\cos(15^\circ)=\frac{\sqrt{6}+\sqrt{2}}{4}$ y $\sin(15^\circ)=\frac{\sqrt{6}-\sqrt{2}}{4}$.
    \begin{align*}
        x' &= -12(\frac{\sqrt{6}+\sqrt{2}}{4}) - 16(\frac{\sqrt{6}-\sqrt{2}}{4}) + 2 = -3(\sqrt{6}+\sqrt{2}) - 4(\sqrt{6}-\sqrt{2}) + 2 \\
           &= -7\sqrt{6}+\sqrt{2}+2 \\
        y' &= -12(\frac{\sqrt{6}-\sqrt{2}}{4}) + 16(\frac{\sqrt{6}+\sqrt{2}}{4}) - 2 = -3(\sqrt{6}-\sqrt{2}) + 4(\sqrt{6}+\sqrt{2}) - 2 \\
           &= \sqrt{6}+7\sqrt{2}-2
    \end{align*}
    \paragraph{Respuesta:} $\mathbf{A'(2+\sqrt{2}-7\sqrt{6}, -2+7\sqrt{2}+\sqrt{6})}$.
\end{enumerate}

\subsection{Ejercicio 8}
\textbf{Problema:} Determinar el punto original A, dadas su imagen A', el centro C y el ángulo $\theta$.
\textbf{Solución:} Para encontrar A, debemos aplicar la rotación inversa a A', es decir, rotar A' alrededor de C con un ángulo de $-\theta$.
\begin{enumerate}[label=\alph*)]
    \item $C(-1,1), A'(-1-2\sqrt{2}, 1-\sqrt{2}), \theta=-45^\circ$. Rotamos A' por $-\theta = +45^\circ$.
    Coordenadas relativas: $(x'-h)=-2\sqrt{2}, (y'-k)=-\sqrt{2}$.
    $x = -2\sqrt{2}\cos(45^\circ) - (-\sqrt{2})\sin(45^\circ) - 1 = -2\sqrt{2}(\frac{\sqrt{2}}{2}) + \sqrt{2}(\frac{\sqrt{2}}{2}) - 1 = -2+1-1 = -2$.
    $y = -2\sqrt{2}\sin(45^\circ) + (-\sqrt{2})\cos(45^\circ) + 1 = -2\sqrt{2}(\frac{\sqrt{2}}{2}) - \sqrt{2}(\frac{\sqrt{2}}{2}) + 1 = -2-1+1 = -2$.
    \paragraph{Respuesta:} $\mathbf{A(-2,-2)}$.
    
    \item $C(1,0), A'(1,-3\sqrt{2}), \theta=225^\circ$. Rotamos A' por $-\theta = -225^\circ$, que es coterminal con $135^\circ$.
    Coordenadas relativas: $(x'-h)=0, (y'-k)=-3\sqrt{2}$.
    $x = 0\cdot\cos(135^\circ) - (-3\sqrt{2})\sin(135^\circ) + 1 = 3\sqrt{2}(\frac{\sqrt{2}}{2}) + 1 = 3+1 = 4$.
    $y = 0\cdot\sin(135^\circ) + (-3\sqrt{2})\cos(135^\circ) + 0 = -3\sqrt{2}(-\frac{\sqrt{2}}{2}) = 3$.
    \paragraph{Respuesta:} $\mathbf{A(4,3)}$.
\end{enumerate}

\subsection{Ejercicio 9}
\textbf{Problema:} Determinar el ángulo $\theta$ que rota A a A' con respecto a C.
\textbf{Solución:} Primero se trasladan los puntos para que el centro de rotación sea el origen: $A_t = A-C$ y $A'_t = A'-C$. Luego, se puede usar el producto punto $\vec{u} \cdot \vec{v} = |\vec{u}||\vec{v}|\cos\theta$.
\begin{enumerate}[label=\alph*)]
    \item $A(\sqrt{5},1), A'(1,\sqrt{5}), C(0,0)$. Los puntos ya están centrados.
    $A \cdot A' = \sqrt{5}\cdot 1 + 1\cdot\sqrt{5} = 2\sqrt{5}$.
    $|A| = \sqrt{(\sqrt{5})^2+1^2} = \sqrt{6}$. $|A'| = \sqrt{1^2+(\sqrt{5})^2} = \sqrt{6}$.
    $\cos\theta = \frac{2\sqrt{5}}{(\sqrt{6})(\sqrt{6})} = \frac{2\sqrt{5}}{6} = \frac{\sqrt{5}}{3}$.
    \paragraph{Respuesta:} $\theta = \arccos\left(\frac{\sqrt{5}}{3}\right) \approx 41.81^\circ$.
    
    \item $A(4,5), A'(5,-2), C(1,1)$. Trasladamos al origen: $A_t(3,4), A'_t(4,-3)$.
    $A_t \cdot A'_t = 3\cdot 4 + 4\cdot(-3) = 12-12=0$.
    Como el producto punto es cero, los vectores son ortogonales. La rotación es de $\pm 90^\circ$. Al observar la posición (del cuadrante I al IV), se concluye que la rotación es en sentido horario.
    \paragraph{Respuesta:} $\theta = -90^\circ$ o $270^\circ$.
\end{enumerate}

\subsection{Ejercicio 10}
\textbf{Problema:} Demostrar que al rotar $A(\sqrt{2}, \sqrt{2})$ por $15^\circ$, su imagen corresponde a la expresión dada.
\textbf{Solución:} Se resolverá en dos partes. Primero, calculando la rotación directamente. Segundo, simplificando la expresión dada.
\paragraph{Parte 1: Cálculo directo de la rotación.}
La fórmula de rotación es $A'(x\cos\theta - y\sin\theta, x\sin\theta + y\cos\theta)$. Se usan las identidades de resta de ángulos para $15^\circ=45^\circ-30^\circ$:
$\cos(15^\circ) = \frac{\sqrt{6}+\sqrt{2}}{4}$ y $\sin(15^\circ) = \frac{\sqrt{6}-\sqrt{2}}{4}$.
Aplicando la rotación:
\begin{align*}
	x' &= \sqrt{2}\left(\frac{\sqrt{6} + \sqrt{2}}{4}\right) - \sqrt{2}\left(\frac{\sqrt{6} - \sqrt{2}}{4}\right) = \frac{\sqrt{12} + 2 - \sqrt{12} + 2}{4} = \frac{4}{4} = 1 \\
	y' &= \sqrt{2}\left(\frac{\sqrt{6} - \sqrt{2}}{4}\right) + \sqrt{2}\left(\frac{\sqrt{6} + \sqrt{2}}{4}\right) = \frac{\sqrt{12} - 2 + \sqrt{12} + 2}{4} = \frac{2\sqrt{12}}{4} = \sqrt{3}
\end{align*}
El punto rotado es $A'(1, \sqrt{3})$.
\paragraph{Parte 2: Simplificación de la expresión dada.}
La coordenada $x$ de la expresión es $X = \sqrt{1 + \frac{\sqrt{3}}{2}} - \sqrt{1 - \frac{\sqrt{3}}{2}}$. Al elevar al cuadrado, $X^2 = (1 + \frac{\sqrt{3}}{2}) - 2\sqrt{1-\frac{3}{4}} + (1 - \frac{\sqrt{3}}{2}) = 2 - 2(\frac{1}{2}) = 1$, por lo tanto $X=1$.
La coordenada $y$ es $Y = \sqrt{1 + \frac{\sqrt{3}}{2}} + \sqrt{1 - \frac{\sqrt{3}}{2}}$. Al elevar al cuadrado, $Y^2 = (1 + \frac{\sqrt{3}}{2}) + 2\sqrt{1-\frac{3}{4}} + (1 - \frac{\sqrt{3}}{2}) = 2 + 2(\frac{1}{2}) = 3$, por lo tanto $Y=\sqrt{3}$.
Como ambos métodos dan $(1, \sqrt{3})$, la proposición queda demostrada.

\subsection{Ejercicio 11}
\textbf{Problema:} Determinar A, dadas A' y $\theta$ (rotación respecto al origen).
\textbf{Solución:} Para encontrar el punto original A, se debe rotar el punto imagen A' por el ángulo opuesto, $-\theta$.
\begin{enumerate}[label=\alph*)]
    \item $A'(\sqrt{2},-\sqrt{6}), \theta=2250^\circ$. Primero, simplificamos el ángulo: $2250^\circ = 6 \times 360^\circ + 90^\circ$. La rotación efectiva es de $90^\circ$. Por lo tanto, rotamos A' por $-90^\circ$.
    $x = \sqrt{2}\cos(-90^\circ) - (-\sqrt{6})\sin(-90^\circ) = \sqrt{2}(0) + \sqrt{6}(-1) = -\sqrt{6}$.
    $y = \sqrt{2}\sin(-90^\circ) + (-\sqrt{6})\cos(-90^\circ) = \sqrt{2}(-1) - \sqrt{6}(0) = -\sqrt{2}$.
    \paragraph{Respuesta:} $\mathbf{A(-\sqrt{6}, -\sqrt{2})}$.
    
    \item $A'(\phi^2, \sqrt{\pi}), \theta=-1080^\circ$. Simplificamos el ángulo: $-1080^\circ = -3 \times 360^\circ$. La rotación efectiva es de $0^\circ$. La rotación inversa también es de $0^\circ$, por lo que el punto no se mueve.
    \paragraph{Respuesta:} $A=A'=\mathbf{(\phi^2, \sqrt{\pi})}$.
\end{enumerate}

\subsection{Ejercicio 12}
\textbf{Problema:} Cuestiones sobre un reloj análogo.
\textbf{Solución:} Tasas de movimiento: horario $0.5^\circ/\text{min}$, minutero $6^\circ/\text{min}$.
\begin{enumerate}[label=\alph*)]
    \item Ángulo barrido por el horario entre 13:00 y 14:15: El tiempo transcurrido es de 75 minutos.
    Ángulo = $75 \text{ min} \times 0.5^\circ/\text{min} = \mathbf{37.5^\circ}$.
    
    \item Ángulo entre punteros a las 14:15: Medimos los ángulos desde la posición de las 12 en sentido horario.
    Posición del minutero: $15 \text{ min} \times 6^\circ/\text{min} = 90^\circ$.
    Posición del horario: $2 \text{ horas} \times 30^\circ/\text{h} + 15 \text{ min} \times 0.5^\circ/\text{min} = 60^\circ + 7.5^\circ = 67.5^\circ$.
    Diferencia: $|90^\circ - 67.5^\circ| = \mathbf{22.5^\circ}$.
    
    \item Coordenadas (con 12 en eje Y+, 3 en eje X+):
    A las 13:00 (1:00), el horario forma $30^\circ$ con el eje Y. Su ángulo en posición estándar es $90^\circ - 30^\circ = 60^\circ$. Coordenadas: $\mathbf{(L\cos60^\circ, L\sin60^\circ) = (L/2, L\sqrt{3}/2)}$.
    A las 14:15, la posición del horario es $67.5^\circ$ desde las 12. Su ángulo en posición estándar es $90^\circ - 67.5^\circ = 22.5^\circ$. Coordenadas: $\mathbf{(L\cos22.5^\circ, L\sin22.5^\circ)}$.
\end{enumerate}

\section{Simetrías}

\subsection{Ejercicio 13}
\textbf{Problema:} Simetría axial del triángulo $A(2,3), B(-1,-3), C(0,6)$ con respecto a $y = -\frac{2}{3}x$.
\textbf{Solución:} Usamos la fórmula de reflexión $A' = (\frac{1-m^2}{1+m^2}x_0 + \frac{2m}{1+m^2}y_0, \frac{2m}{1+m^2}x_0 - \frac{1-m^2}{1+m^2}y_0)$, con $m=-2/3$.
Calculamos los factores constantes: $1+m^2=13/9$, $1-m^2=5/9$, $2m=-4/3$.
$\frac{1-m^2}{1+m^2} = \frac{5}{13}$, $\frac{2m}{1+m^2} = -\frac{12}{13}$.
\begin{itemize}
    \item $A' = (\frac{5}{13}(2) - \frac{12}{13}(3), -\frac{12}{13}(2) - \frac{5}{13}(3)) = (-\frac{26}{13}, -\frac{39}{13}) = \mathbf{(-2, -3)}$.
    \item $B' = (\frac{5}{13}(-1) - \frac{12}{13}(-3), -\frac{12}{13}(-1) - \frac{5}{13}(-3)) = (\frac{31}{13}, \frac{27}{13})$.
    \item $C' = (\frac{5}{13}(0) - \frac{12}{13}(6), -\frac{12}{13}(0) - \frac{5}{13}(6)) = \mathbf{(-\frac{72}{13}, -\frac{30}{13})}$.
\end{itemize}

\subsection{Ejercicio 14}
\textbf{Problema:} Calcular el área encerrada por los triángulos ABC y A'B'C', donde A'B'C' es el simétrico de ABC respecto a $x=-3$.
\textbf{Solución:} La simetría axial es una isometría, por lo que el área del triángulo reflejado A'B'C' es idéntica a la del original ABC. El área total encerrada es el doble del área de uno de los triángulos.
Vértices: $A(-2,-4), B(-2,6), C(-6,2)$.
La base puede ser el segmento vertical AB. Longitud de la base: $|6 - (-4)| = 10$.
La altura es la distancia horizontal del vértice C a la recta $x=-2$. Altura: $|-6 - (-2)| = 4$.
Área(ABC) = $\frac{1}{2} \times \text{base} \times \text{altura} = \frac{1}{2} \times 10 \times 4 = 20$.
\paragraph{Respuesta:} El área total encerrada es $2 \times 20 = \mathbf{40}$ unidades cuadradas.

\subsection{Ejercicio 15}
\textbf{Problema:} Calcular ángulos del triángulo ABC formado por simetrías sucesivas.
\textbf{Solución:} Primero, encontramos los vértices.
$A(2,1)$.
$B$ es el reflejo de A respecto a $y=x$, entonces $B(1,2)$.
$C$ es el reflejo de B respecto a $y=-x$, entonces $C(-2,-1)$.
Calculamos el cuadrado de las longitudes de los lados para verificar si es un triángulo rectángulo:
$AB^2 = (1-2)^2 + (2-1)^2 = (-1)^2 + 1^2 = 2$.
$BC^2 = (-2-1)^2 + (-1-2)^2 = (-3)^2 + (-3)^2 = 9+9 = 18$.
$AC^2 = (-2-2)^2 + (-1-1)^2 = (-4)^2 + (-2)^2 = 16+4 = 20$.
Como $AB^2+BC^2=2+18=20=AC^2$, por el Teorema de Pitágoras, el triángulo es rectángulo en B.
\paragraph{Respuesta:} $\angle B = 90^\circ$. Los otros ángulos se calculan por trigonometría: $\cos(\angle A) = \frac{AB}{AC} = \frac{\sqrt{2}}{\sqrt{20}} \implies \angle A \approx 71.57^\circ$, y $\angle C \approx 18.43^\circ$.

\subsection{Ejercicio 16}
\textbf{Problema:} Coordenadas de $A'$ y $A''$ por simetrías sucesivas.
\begin{enumerate}[label=\alph*)]
    \item $A(-4,2)$. $L_1: y=x \implies A'(2,-4)$. $L_2: y=-x \implies A''$ (reflejo de $A'$) $= \mathbf{(4,-2)}$.
    \item $A(-4,2)$. $L_1: y=x/3, L_2: y=-2/3 x$. Se aplican las fórmulas del Ej. 13.
    Para $L_1 (m=1/3)$: Factores: $\frac{1-m^2}{1+m^2}=\frac{8/9}{10/9}=\frac{4}{5}$, $\frac{2m}{1+m^2}=\frac{2/3}{10/9}=\frac{3}{5}$.
    $A' = (\frac{4}{5}(-4) + \frac{3}{5}(2), \frac{3}{5}(-4) - \frac{4}{5}(2)) = \mathbf{(-\frac{10}{5}, -\frac{20}{5}) = (-2, -4)}$.
    Para $L_2 (m=-2/3)$ aplicada a $A'(-2,-4)$: Factores: $\frac{5}{13}$ y $-\frac{12}{13}$.
    $A'' = (\frac{5}{13}(-2) - \frac{12}{13}(-4), -\frac{12}{13}(-2) - \frac{5}{13}(-4)) = \mathbf{(\frac{38}{13}, \frac{44}{13})}$.
\end{enumerate}

\subsection{Ejercicio 17}
\textbf{Problema:} Determinar el centro de una simetría central.
\textbf{Solución:} El centro de simetría C es el punto medio del segmento que une A y A'. Se calcula con la fórmula $C = \left(\frac{x_A+x_{A'}}{2}, \frac{y_A+y_{A'}}{2}\right)$.
\begin{enumerate}[label=\alph*)]
    \item $A(-1,3), A'(3,-3) \implies C = \left(\frac{-1+3}{2}, \frac{3-3}{2}\right) = \mathbf{(1,0)}$.
    \item $B(-1/2, 3/2), B'(9/4, 11/4) \implies C = \left(\frac{-1/2+9/4}{2}, \frac{3/2+11/4}{2}\right) = \left(\frac{7/4}{2}, \frac{17/4}{2}\right) = \mathbf{(7/8, 17/8)}$.
\end{enumerate}

\end{itemize}

\section{Homotecias}

\subsection{Ejercicio 19}
\textbf{Problema:} ¿Cuándo una homotecia es equivalente a una simetría central?
\textbf{Solución:} Una simetría central con respecto a un punto C invierte la posición de cada punto a través del centro. La fórmula de homotecia es $A' = C + k(A-C)$. Si $k=-1$, la fórmula se convierte en $A' = C - (A-C) = 2C - A$, que es precisamente la fórmula para la simetría central. Por lo tanto, una homotecia es equivalente a una simetría central cuando su razón es $\mathbf{k=-1}$.

\subsection{Ejercicio 20}
\textbf{Problema:} Coordenadas de P'Q'R' simétrico a PQR con respecto a C para varias razones k.
\textbf{Solución:} Se utiliza la fórmula de homotecia $A' = (h(1-k)+kx, k(1-k)+ky)$.
Dados: $P(-1,-1), Q(1,-2), R(0,1), C(-1,2)$.
\begin{enumerate}[label=\alph*)]
    \item k=5/2: $1-k = -3/2$.
    $P' = (-1(-\frac{3}{2}) + \frac{5}{2}(-1), 2(-\frac{3}{2}) + \frac{5}{2}(-1)) = (\frac{3}{2}-\frac{5}{2}, -3-\frac{5}{2}) = \mathbf{(-1, -5.5)}$.
    $Q' = (-1(-\frac{3}{2}) + \frac{5}{2}(1), 2(-\frac{3}{2}) + \frac{5}{2}(-2)) = (\frac{3}{2}+\frac{5}{2}, -3-5) = \mathbf{(4, -8)}$.
    $R' = (-1(-\frac{3}{2}) + \frac{5}{2}(0), 2(-\frac{3}{2}) + \frac{5}{2}(1)) = (\frac{3}{2}, -3+\frac{5}{2}) = \mathbf{(1.5, -0.5)}$.
    \item k=1: La transformación es la identidad. $\mathbf{P'=P, Q'=Q, R'=R}$.
    \item k=-1 (Simetría Central): $1-k = 2$.
    $P' = (-1(2) - 1(-1), 2(2) - 1(-1)) = \mathbf{(-1,5)}$.
    $Q' = (-1(2) - 1(1), 2(2) - 1(-2)) = \mathbf{(-3,6)}$.
    $R' = (-1(2) - 1(0), 2(2) - 1(1)) = \mathbf{(-2,3)}$.
\end{enumerate}

\subsection{Ejercicio 21}
\textbf{Problema:} Calcular las áreas de los triángulos P'Q'R' obtenidos.
\textbf{Solución:} La relación de áreas en una homotecia es Área(Imagen) = $k^2 \cdot$ Área(Original).
Primero, calculamos el área del triángulo original $PQR$ con $P(-1,-1), Q(1,-2), R(0,1)$ usando la fórmula del lazo:
Área = $\frac{1}{2} |(x_P y_Q + x_Q y_R + x_R y_P) - (y_P x_Q + y_Q x_R + y_R x_P)|$
Área = $\frac{1}{2} |((-1)(-2) + 1(1) + 0(-1)) - ((-1)(1) + (-2)(0) + 1(-1))| = \frac{1}{2} |(2+1) - (-1-1)| = \frac{5}{2} = 2.5$.
\begin{itemize}
    \item[a)] k=5/2: Área = $(5/2)^2 \cdot 2.5 = 6.25 \cdot 2.5 = \mathbf{15.625}$.
    \item[b)] k=1: Área = $1^2 \cdot 2.5 = \mathbf{2.5}$.
    \item[c)] k=2/3: Área = $(2/3)^2 \cdot 2.5 = \frac{4}{9} \cdot \frac{5}{2} = \mathbf{10/9}$.
    \item[d)] k=-0.3: Área = $(-0.3)^2 \cdot 2.5 = 0.09 \cdot 2.5 = \mathbf{0.225}$.
    \item[e)] k=-1: Área = $(-1)^2 \cdot 2.5 = \mathbf{2.5}$.
    \item[f)] k=-2.16: Área = $(-2.16)^2 \cdot 2.5 = 4.6656 \cdot 2.5 = \mathbf{11.664}$.
\end{itemize}

\subsection{Ejercicio 22}
\textbf{Problema:} Demostrar la relación de proporcionalidad en una homotecia.
\textbf{Solución:}
\paragraph{Principio fundamental:} La definición de homotecia establece que el centro $C$, el punto original $A$ y el punto imagen $A'$ son colineales. 
\paragraph{Ecuación vectorial:} Este principio se traduce en que el vector que va del centro a la imagen ($\vec{CA'}$) es un múltiplo escalar del vector que va del centro al punto original ($\vec{CA}$). El escalar es la razón de homotecia, $k$.
\[ \vec{CA'} = k \cdot \vec{CA} \]
\paragraph{Expresión en componentes:} Expresamos los vectores usando las coordenadas de los puntos $A(x_0,y_0)$, $A'(x_1,y_1)$ y $C(h,k)$.
\[ (x_1 - h, y_1 - k) = k \cdot (x_0 - h, y_0 - k) \]
\paragraph{Despeje de la razón k:} Esta igualdad vectorial se descompone en dos ecuaciones escalares, una para el eje X y otra para el eje Y:
\begin{align*}
	x_1 - h & = k(x_0 - h) \implies k = \frac{x_1 - h}{x_0 - h} \\
	y_1 - k & = k(y_0 - k) \implies k = \frac{y_1 - k}{y_0 - k}
\end{align*}
\paragraph{Conclusión:} Como ambas expresiones son iguales a la misma constante $k$, deben ser iguales entre sí, lo que demuestra la relación solicitada:
\[ \frac{x_1 - h}{x_0 - h} = \frac{y_1 - k}{y_0 - k} \]

\subsection{Ejercicio 23}
\textbf{Problema:} Determinar $k$ para la homotecia dada.
\textbf{Solución:} Se nos dan $A(-2,2), C(-2,-1), A'(-2,-7)$. La fórmula para la razón, demostrada en el ejercicio anterior, es $k = \frac{x'-h}{x-h}$ o $k = \frac{y'-k}{y-k}$.
Intentamos con las abscisas (coordenadas x):
\[ k = \frac{-2 - (-2)}{-2 - (-2)} = \frac{0}{0} \]
Obtenemos una indeterminación. Esto ocurre porque los tres puntos son colineales sobre una recta vertical ($x=-2$), lo que significa que no hay un "estiramiento" horizontal que medir. Por lo tanto, es necesario usar las ordenadas (coordenadas y):
\[ k = \frac{y_1 - k}{y_0 - k} = \frac{-7 - (-1)}{2 - (-1)} = \frac{-6}{3} = -2 \]
\paragraph{Respuesta:} La razón de homotecia es $\mathbf{k=-2}$.

\subsection{Ejercicio 24}
\textbf{Problema:} Dado el rombo PQRS con $P(-1,-3)$, $Q(0,-1)$ y el lado $\overline{RQ}$ paralelo al eje X.

\begin{enumerate}[label=\alph*)]
    \item \textbf{Determinar coordenadas de R y S:}
    La condición $\overline{RQ} \parallel \text{Eje X}$ implica que $y_R = y_Q = -1$.
    En un rombo, todos los lados tienen la misma longitud. Calculamos la longitud del lado PQ:
    $|\vec{PQ}| = \sqrt{(0 - (-1))^2 + (-1 - (-3))^2} = \sqrt{1^2 + 2^2} = \sqrt{5}$.
    La longitud del lado QR debe ser la misma. Como $R(x_R, -1)$, la longitud es:
    $|\vec{QR}| = \sqrt{(x_R - 0)^2 + (-1 - (-1))^2} = \sqrt{x_R^2} = |x_R|$.
    Por lo tanto, $|x_R| = \sqrt{5}$. Asumiendo por la figura que R está a la derecha de Q, $x_R = \sqrt{5}$.
    $\implies \mathbf{R(\sqrt{5}, -1)}$.
    Para un paralelogramo, se cumple la relación vectorial $\vec{P}+\vec{R} = \vec{Q}+\vec{S}$, de donde $S = P + R - Q$:
    $S = (-1, -3) + (\sqrt{5}, -1) - (0, -1) = (\sqrt{5}-1, -3)$.
    $\implies \mathbf{S(\sqrt{5}-1, -3)}$.

    \item \textbf{Coordenadas de la imagen con k=2 y C(2,0):}
    Usamos la fórmula $A' = (h(1-k)+kx, k(1-k)+ky)$.
    $P' = (2(1-2)+2(-1), 0(1-2)+2(-3)) = \mathbf{(-4, -6)}$.
    $Q' = (2(1-2)+2(0), 0(1-2)+2(-1)) = \mathbf{(-2, -2)}$.
    $R' = (2(1-2)+2(\sqrt{5}), 0(1-2)+2(-1)) = \mathbf{(-2+2\sqrt{5}, -2)}$.
    $S' = (2(1-2)+2(\sqrt{5}-1), 0(1-2)+2(-3)) = \mathbf{(-4+2\sqrt{5}, -6)}$.

    \item \textbf{Coordenadas de la imagen con k=-1/2 y C(-2,-2):}
    Usamos la misma fórmula. El factor $1-k = 1 - (-1/2) = 3/2$.
    $P' = (-2(\frac{3}{2}) - \frac{1}{2}(-1), -2(\frac{3}{2}) - \frac{1}{2}(-3)) = \mathbf{(-\frac{5}{2}, -\frac{3}{2})}$.
    $Q' = (-2(\frac{3}{2}) - \frac{1}{2}(0), -2(\frac{3}{2}) - \frac{1}{2}(-1)) = \mathbf{(-3, -\frac{5}{2})}$.
    $R' = (-2(\frac{3}{2}) - \frac{1}{2}(\sqrt{5}), -2(\frac{3}{2}) - \frac{1}{2}(-1)) = \mathbf{(-\frac{6+\sqrt{5}}{2}, -\frac{5}{2})}$.
    $S' = (-2(\frac{3}{2}) - \frac{1}{2}(\sqrt{5}-1), -2(\frac{3}{2}) - \frac{1}{2}(-3)) = \mathbf{(-\frac{5+\sqrt{5}}{2}, -\frac{3}{2})}$.
\end{enumerate}
